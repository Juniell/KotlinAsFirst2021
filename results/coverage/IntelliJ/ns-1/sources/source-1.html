281 // реализовать бинарный поиск, работать будет за n*log(n)
282 fun findSumOfTwo(list: List<Int>, number: Int): Pair<Int, Int> {
283     // создаем две копии изначального списка, потому что один мы отсортируем, чтобы работал бинарный поиск
284     val newList = arrayListOf<Int>()
285     val oldList = arrayListOf<Int>()
286 
287     var firstIndex = -1
288     var secondIndex = -1
289     list.forEach {
290         newList.add(it)
291         oldList.add(it)
292     }
293 
294     newList.sort()
295 
296     for (i in newList.indices) {
297         val findNumber = number - newList[i]
298         val currentNumber = newList[i]
299 
300         var isFindNumber = false
301 
302         var leftBorder = 0
303         var rightBorder = newList.size - 1
304 
305         // сам бинарный поиск, вернет true, если findNumber есть в списке с учетом того,
306         // что индекс findNumber не совпадает с текущим индексом элемента
307         while (leftBorder <= rightBorder) {
308             val middle = (leftBorder + rightBorder) / 2
309             if (findNumber == newList[middle] && middle != i) {
310                 isFindNumber = true
311                 break
312             }
313             if (findNumber < newList[middle]) rightBorder = middle - 1
314             else leftBorder = middle + 1
315         }
316 
317         if (isFindNumber) {
318             // т.к элемент был найден, то можно пройтись по старому списку и найти значения индексов
319             for (j in 0 until oldList.size) {
320                 if (oldList[j] == currentNumber && j != secondIndex && firstIndex == -1) firstIndex = j
321                 if (oldList[j] == findNumber && j != firstIndex && secondIndex == -1) secondIndex = j
322                 if (firstIndex > 0 && secondIndex > 0) break
323             }
324             return Pair(firstIndex, secondIndex)
325         }
326     }
327     return Pair(-1, -1)
328 }

352 fun bagPacking(treasures: Map<String, Pair<Int, Int>>, capacity: Int): Set<String> {
353     // списки с массой, ценной, названием каждого предмета
354     // например, у j-ого предмета название names[j], масса masses[j], стоимость values[j]
355     val masses = arrayListOf<Int>()
356     val values = arrayListOf<Int>()
357     val names = arrayListOf<String>()
358     // добавил нулевые элементы, чтобы не путаться с индексами
359     masses.add(0)
360     values.add(0)
361     names.add("")
362     for ((key, value) in treasures) {
363         masses.add(value.first)
364         values.add(value.second)
365         names.add(key)
366     }
367     // матрица с максимальной ценной на каждую массу и кол-во предметов, где масса от 0..capacity
368     val matrix = Array(capacity + 1) { IntArray(values.size + 1) }
369     // матрица со списком для каждой массы и кол-ва предметов(эта матрица нужна, чтобы хранить на каждом этапе список товаров,
370     // чтобы по итогу вывести предметы)
371     val matrixNames = Array(capacity + 1) { Array(values.size + 1) { "" } }
372     for (i in 0..capacity) {
373         for (j in 0..values.size) {
374             matrix[i][j] = 0
375         }
376     }
377 
378     // вся логика
379     for (i in 1 until capacity + 1) {
380         for (j in 1 until values.size) {
381             if (i >= masses[j] && (matrix[i - masses[j]][j - 1] + values[j]) > matrix[i][j - 1]) {
382                 matrix[i][j] = values[j] + matrix[i - masses[j]][j - 1]
383                 matrixNames[i][j] = matrixNames[i - masses[j]][j - 1]
384                 matrixNames[i][j] += "${names[j]} "
385             } else {
386                 matrix[i][j] = matrix[i][j - 1]
387                 matrixNames[i][j] = matrixNames[i][j - 1]
388             }
389         }
390     }
391 
392     val lenLastString = matrixNames[capacity][values.size - 1].length
393     if (lenLastString != 0)
394         return matrixNames[capacity][values.size - 1].substring(0, lenLastString - 1).split(" ").toSet()
395     return setOf()
396 }
