27 fun alignFile(inputName: String, lineLength: Int, outputName: String) {
28     val writer = File(outputName).bufferedWriter()
29     var currentLineLength = 0
30     fun append(word: String) {
31         if (currentLineLength > 0) {
32             if (word.length + currentLineLength >= lineLength) {
33                 writer.newLine()
34                 currentLineLength = 0
35             } else {
36                 writer.write(" ")
37                 currentLineLength++
38             }
39         }
40         writer.write(word)
41         currentLineLength += word.length
42     }
43     for (line in File(inputName).readLines()) {
44         if (line.isEmpty()) {
45             writer.newLine()
46             if (currentLineLength > 0) {
47                 writer.newLine()
48                 currentLineLength = 0
49             }
50             continue
51         }
52         for (word in line.split(Regex("\\s+"))) {
53             append(word)
54         }
55     }
56     writer.close()
57 }
